# v0.5 Safe Fix Mode Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Fill the `fix` command so it scans readiness notes and applies safe patches to 4 classes of source-repo issues, with dry-run by default and `--apply` to write.

**Architecture:** New `src/fix/` module — fixers are pure functions `(brief, fileContents) => Fix[]`, chain-merged via a mutable fileContents Map so multiple fixers editing the same file produce one final Fix. Zero-dep LCS diff renderer. Scanner/render/MCP untouched.

**Tech Stack:** TypeScript (ESM, `tsc`), Node ≥ 20, `node:test`, Node stdlib only.

**Reference spec:** `docs/superpowers/specs/2026-06-19-v0.5-safe-fix-mode-design.md`

```typescript
/**
 * Renders a unified diff (git-style header + @@ hunks + context/removed/added
 * lines) between two string contents. Zero dependencies; uses LCS.
 */
export function renderDiff(
  relPath: string,
  originalContent: string,
  patchedContent: string,
): string {
  const a = originalContent.split("\n");
  const b = patchedContent.split("\n");

  // Trailing empty string from final newline: keep for indexing but it
  // represents the empty segment after the last \n. We keep all segments.
  const m = a.length;
  const n = b.length;

  // LCS DP table (m+1 x n+1). For small files this is fine.
  const dp: number[][] = Array.from({ length: m + 1 }, () =>
    new Array<number>(n + 1).fill(0),
  );
  for (let i = m - 1; i >= 0; i--) {
    for (let j = n - 1; j >= 0; j--) {
      if (a[i] === b[j]) {
        dp[i][j] = dp[i + 1][j + 1] + 1;
      } else {
        dp[i][j] = Math.max(dp[i + 1][j], dp[i][j + 1]);
      }
    }
  }

  // Walk forward producing op list.
  type Op = { kind: "eq" | "del" | "add"; line: string };
  const ops: Op[] = [];
  let i = 0;
  let j = 0;
  while (i < m && j < n) {
    if (a[i] === b[j]) {
      ops.push({ kind: "eq", line: a[i] });
      i++;
      j++;
    } else if (dp[i + 1][j] >= dp[i][j + 1]) {
      ops.push({ kind: "del", line: a[i] });
      i++;
    } else {
      ops.push({ kind: "add", line: b[j] });
      j++;
    }
  }
  while (i < m) {
    ops.push({ kind: "del", line: a[i] });
    i++;
  }
  while (j < n) {
    ops.push({ kind: "add", line: b[j] });
    j++;
  }

  // Group into hunks. A hunk starts at the first non-eq op and extends with
  // CONTEXT surrounding eq runs. We emit one hunk covering all changes with
  // 3 lines of context (clamped to file bounds).
  const CONTEXT = 3;
  const changeIndices: number[] = [];
  for (let k = 0; k < ops.length; k++) {
    if (ops[k].kind !== "eq") changeIndices.push(k);
  }
  if (changeIndices.length === 0) return "";

  const hunkStart = Math.max(0, changeIndices[0] - CONTEXT);
  const hunkEnd = Math.min(ops.length - 1, changeIndices[changeIndices.length - 1] + CONTEXT);

  const lines: string[] = [];
  lines.push(`--- a/${relPath}`);
  lines.push(`+++ b/${relPath}`);

  // Compute hunk header line ranges in 1-indexed original/patched coords.
  let oldStart = 0;
  let oldCount = 0;
  let newStart = 0;
  let newCount = 0;
  let oldLine = 1;
  let newLine = 1;
  for (let k = 0; k < ops.length; k++) {
    if (k === hunkStart) {
      oldStart = oldLine;
      newStart = newLine;
    }
    if (k >= hunkStart && k <= hunkEnd) {
      if (ops[k].kind === "eq") {
        oldLine++;
        newLine++;
        oldCount++;
        newCount++;
      } else if (ops[k].kind === "del") {
        oldLine++;
        oldCount++;
      } else {
        newLine++;
        newCount++;
      }
    } else {
      if (ops[k].kind === "eq") {
        oldLine++;
        newLine++;
      } else if (ops[k].kind === "del") {
        oldLine++;
      } else {
        newLine++;
      }
    }
  }

  lines.push(`@@ -${oldStart},${oldCount} +${newStart},${newCount} @@`);

  for (let k = hunkStart; k <= hunkEnd; k++) {
    const op = ops[k];
    if (op.kind === "eq") lines.push(` ${op.line}`);
    else if (op.kind === "del") lines.push(`-${op.line}`);
    else lines.push(`+${op.line}`);
  }

  return lines.join("\n");
}
```

```typescript
import type { Fix, Fixer, FixerContext } from "./types.js";

/**
 * Patches `packageManager` field value from <declared>@<version> to
 * <lockfileManager>@<version>, preserving version and file formatting.
 */
export const fixPackageManagerField: Fixer = (ctx: FixerContext): Fix[] => {
  const note = ctx.brief.readinessNotes.find((n) =>
    n.startsWith("packageManager field declares "),
  );
  if (!note) return [];

  // Note format: "packageManager field declares X, but lockfile suggests Y."
  const declaredMatch = note.match(/declares (\w+)/);
  const lockfileMatch = note.match(/lockfile suggests (\w+)\./);
  if (!declaredMatch || !lockfileMatch) return [];

  const declared = declaredMatch[1];
  const target = lockfileMatch[1];

  const relPath = "package.json";
  const filePath = `${ctx.root}/${relPath}`;
  const original = ctx.files.get(filePath);
  if (original === undefined) return [];

  // Match:  "packageManager": "declared@version"  OR  "packageManager": "declared"
  const fieldRe = new RegExp(
    `("packageManager"\\s*:\\s*")${declared}(@[^"]*)?(")`,
  );
  const patched = original.replace(fieldRe, (_m, pre, _at, version, post) => {
    return `${pre}${target}${version ?? ""}${post}`;
  });
  if (patched === original) return [];

  ctx.files.set(filePath, patched);
  return [
    {
      filePath,
      relPath,
      certainty: "deterministic",
      note,
      originalContent: original,
      patchedContent: patched,
    },
  ];
};

// Filled in by later tasks.
const ALL_FIXERS: Fixer[] = [
  fixPackageManagerField,
];

/**
 * Runs all fixers in order against a shared mutable fileContents map (so
 * later fixers see earlier fixers' patches — chain-patch). Returns the
 * collected Fix list.
 */
export function runFixers(ctx: FixerContext): Fix[] {
  const fixes: Fix[] = [];
  for (const fixer of ALL_FIXERS) {
    fixes.push(...fixer(ctx));
  }
  return fixes;
}
```

Note: the regex captures group 2 as `@version` (including the `@`), so the replace uses `version ?? ""` for the no-version case.

```typescript
const BARE_SCRIPTS = ["test", "build", "lint", "dev"];

/**
 * Rewrites package-manager script command prefixes in `content` so that every
 * script invocation uses `target` as the manager. install/add/remove/exec/dlx/ci
 * are left untouched.
 */
function rewritePackageManagerCommands(content: string, target: string): string {
  // Build a single pass of replacements. We tokenize on word boundaries for
  // the manager prefix, then decide per-token whether it's a script command.
  //
  // Approach: scan with a regex that matches any of the four managers
  // followed by an optional "run" and a script name, then re-emit using the
  // target manager's syntax.
  const managerRe =
    /\b(npm|pnpm|yarn|bun)(?:\s+(run))?\s+([a-zA-Z0-9:_-]+)/g;

  return content.replace(managerRe, (full, manager: string, run: string | undefined, script: string) => {
    if (PACKAGE_MANAGER_COMMANDS.has(script)) return full;
    if (manager === target) return full;

    // Map to target syntax.
    if (target === "npm") {
      if (BARE_SCRIPTS.includes(script)) return `npm ${script}`;
      return `npm run ${script}`;
    }
    if (target === "pnpm") {
      // pnpm accepts both `pnpm X` and `pnpm run X`; use the shorter form.
      return `pnpm ${script}`;
    }
    if (target === "yarn") {
      return `yarn ${script}`;
    }
    if (target === "bun") {
      return `bun ${script}`;
    }
    return full;
  });
}
```

```typescript
 * Rewrites package-manager command prefixes in CI workflow `run:` lines so
 * they match the lockfile manager. Only `run:` steps are touched; `uses:` and
 * other YAML are preserved.
 */
export const fixCiPackageManagerCommands: Fixer = (ctx: FixerContext): Fix[] => {
  const note = ctx.brief.readinessNotes.find((n) =>
    n.startsWith("GitHub Actions uses ") && n.includes(", but lockfile suggests"),
  );
  if (!note) return [];

  const target = ctx.brief.packageManager;
  if (target === "unknown") return [];

  const fixes: Fix[] = [];
  for (const relPath of ctx.brief.structure.ciWorkflows) {
    const filePath = `${ctx.root}/${relPath}`;
    const original = ctx.files.get(filePath);
    if (original === undefined) continue;

    // Rewrite only the runValue portion of run: lines.
    const patched = original.split("\n").map((line) => {
      const trimmed = line.trim();
      const runValue =
        trimmed.startsWith("- run:")
          ? trimmed.slice("- run:".length).trim()
          : trimmed.startsWith("run:")
            ? trimmed.slice("run:".length).trim()
            : null;
      if (runValue === null) return line;

      const newRunValue = rewritePackageManagerCommands(runValue, target);
      if (newRunValue === runValue) return line;
      return line.replace(runValue, newRunValue);
    }).join("\n");

    if (patched === original) continue;

    ctx.files.set(filePath, patched);
    fixes.push({
      filePath,
      relPath,
      certainty: "deterministic",
      note,
      originalContent: original,
      patchedContent: patched,
    });
  }
  return fixes;
};
```

```typescript
/**
 * Inserts conventional lint/typecheck scripts into package.json when the
 * corresponding readiness note fires. Marked `assumed-standard` because the
 * command assumes a default config.
 */
export const fixMissingStandardScripts: Fixer = (ctx: FixerContext): Fix[] => {
  const relPath = "package.json";
  const filePath = `${ctx.root}/${relPath}`;
  let original = ctx.files.get(filePath);
  if (original === undefined) return [];

  const fixes: Fix[] = [];

  const hasLintNote = ctx.brief.readinessNotes.some(
    (n) => n.startsWith("ESLint config found") && n.includes("no lint script"),
  );
  const hasTypecheckNote = ctx.brief.readinessNotes.some(
    (n) => n.startsWith("tsconfig.json found") && n.includes("no typecheck script"),
  );

  const insertions: { script: string; command: string; note: string }[] = [];
  if (hasLintNote && !/"lint"\s*:/.test(original)) {
    insertions.push({ script: "lint", command: "eslint .", note: "ESLint config found but no lint script in package.json. Consider adding one." });
  }
  if (hasTypecheckNote && !/"typecheck"\s*:/.test(original)) {
    insertions.push({ script: "typecheck", command: "tsc --noEmit", note: "tsconfig.json found but no typecheck script in package.json. Consider adding one." });
  }
  if (insertions.length === 0) return [];

  // Snapshot original once; we may insert multiple scripts.
  const snapshot = original;

  for (const ins of insertions) {
    const before = original;
    original = insertScript(original, ins.script, ins.command);
    if (original !== before) {
      ctx.files.set(filePath, original);
      fixes.push({
        filePath,
        relPath,
        certainty: "assumed-standard",
        note: ins.note,
        originalContent: before,
        patchedContent: original,
      });
    }
  }

  // Collapse: the final Fix should carry snapshot→final. But each push above
  // carries a partial transition. For clean display we replace the list with
  // a single combined Fix when there are multiple insertions.
  if (fixes.length > 1) {
    return [
      {
        filePath,
        relPath,
        certainty: "assumed-standard",
        note: fixes.map((f) => f.note).join(" / "),
        originalContent: snapshot,
        patchedContent: original,
      },
    ];
  }
  return fixes;
};

/**
 * Inserts `"name": "command"` before the closing `}` of the `scripts` block.
 * Uses regex to preserve file formatting. If no scripts block exists, returns
 * content unchanged (we don't fabricate a block).
 */
function insertScript(content: string, name: string, command: string): string {
  // Match the scripts object: "scripts": { ... }. Non-greedy, single block.
  const scriptsBlockRe = /("scripts"\s*:\s*\{)([\s\S]*?)(\})/;
  const match = content.match(scriptsBlockRe);
  if (!match) return content;

  const [, open, body, close] = match;
  // Build the new entry. If body has content, append a comma after the last
  // existing entry; else insert as the sole entry.
  const trimmedBody = body.replace(/\s+$/, "");
  const hasExisting = /\S/.test(trimmedBody);
  const indentMatch = body.match(/\n(\s+)\S/);
  const indent = indentMatch ? indentMatch[1] : "  ";
  const newEntry = `${hasExisting ? "," : ""}\n${indent}"${name}": "${command}"\n`;

  const newBody = `${trimmedBody}${newEntry}`;
  return content.replace(scriptsBlockRe, `${open}${newBody}${close}`);
}
```

```typescript
import { writeFile } from "node:fs/promises";
import type { ApplySummary, Fix } from "./types.js";
import { renderDiff } from "./diff.js";

/**
 * Renders the fix list as labeled diff blocks for dry-run display.
 */
export function renderDiffs(fixes: Fix[]): string {
  return fixes
    .map((fix) => {
      const diff = renderDiff(fix.relPath, fix.originalContent, fix.patchedContent);
      if (!diff) return "";
      return [`[${fix.certainty}] ${fix.relPath}`, `  reason: ${fix.note}`, diff].join(
        "\n",
      );
    })
    .filter((s) => s !== "")
    .join("\n\n");
}

/**
 * Writes each fix's patchedContent to disk. Returns a summary of what changed.
 */
export async function applyFixes(fixes: Fix[]): Promise<ApplySummary> {
  for (const fix of fixes) {
    await writeFile(fix.filePath, fix.patchedContent, "utf8");
  }
  return {
    applied: fixes.length,
    files: fixes.map((f) => f.relPath),
  };
}
```

```typescript
import { readFile, readdir } from "node:fs/promises";
import { join } from "node:path";
import { scanRepository } from "../scanner.js";
import { runFixers } from "./fixers.js";
import { renderDiffs, applyFixes } from "./apply.js";
import type { CliIo, Fix, FixCommandOptions, FixerContext } from "./types.js";

/**
 * Pre-reads all files that fixers might touch into a Map. The Map is mutated
 * by fixers (chain-patch); the original raw contents are snapshotted first so
 * we can compute per-file original→patched diffs.
 */
async function loadFiles(root: string): Promise<{
  original: Map<string, string>;
  current: Map<string, string>;
}> {
  const candidates = [
    "package.json",
    "README.md",
    ".github/workflows",
  ];
  const current = new Map<string, string>();
  const original = new Map<string, string>();

  for (const candidate of candidates) {
    const abs = join(root, candidate);
    if (candidate === ".github/workflows") {
      let entries: string[];
      try {
        entries = await readdir(abs);
      } catch {
        continue;
      }
      for (const entry of entries.filter((f) => f.endsWith(".yml") || f.endsWith(".yaml"))) {
        const wfPath = join(abs, entry);
        const content = await safeRead(wfPath);
        if (content !== null) {
          current.set(wfPath, content);
          original.set(wfPath, content);
        }
      }
    } else {
      const content = await safeRead(abs);
      if (content !== null) {
        current.set(abs, content);
        original.set(abs, content);
      }
    }
  }
  return { original, current };
}

async function safeRead(path: string): Promise<string | null> {
  try {
    return await readFile(path, "utf8");
  } catch {
    return null;
  }
}

export async function runFixCommand(
  root: string,
  options: FixCommandOptions,
  io: CliIo,
): Promise<number> {
  const brief = await scanRepository(root);
  const { original, current } = await loadFiles(root);

  const ctx: FixerContext = { brief, files: current, root };
  const rawFixes = runFixers(ctx);

  // Collapse chain-patched files into one Fix per file using original snapshot.
  const fixesByFile = new Map<string, Fix>();
  for (const fix of rawFixes) {
    const existing = fixesByFile.get(fix.filePath);
    if (existing) {
      // Merge: extend the note, keep earliest original, update patched to latest.
      fixesByFile.set(fix.filePath, {
        ...existing,
        note: `${existing.note} / ${fix.note}`,
        patchedContent: fix.patchedContent,
      });
    } else {
      // Use the ORIGINAL snapshot (pre-any-fixer) for clean diff.
      fixesByFile.set(fix.filePath, {
        ...fix,
        originalContent: original.get(fix.filePath) ?? fix.originalContent,
      });
    }
  }
  const fixes = [...fixesByFile.values()];

  if (fixes.length === 0) {
    io.stdout("No fixes needed. Your repo looks clean.");
    return 0;
  }

  if (!options.apply) {
    io.stdout(renderDiffs(fixes));
    io.stdout("");
    io.stdout("Dry-run: no files changed. Run with --apply to write.");
    return 0;
  }

  const summary = await applyFixes(fixes);
  io.stdout(
    `Applied ${summary.applied} fix(es) to ${summary.files.length} file(s): ${summary.files.join(", ")}`,
  );
  io.stdout("Review with: git diff");
  io.stdout("Rollback with: git checkout -- <file>");
  return 0;
}
```
